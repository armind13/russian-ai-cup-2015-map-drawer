#include <fstream>
if (1 == world.getTick())
    {
        ofstream mapFile;
        mapFile.open("map.txt");
        auto mapTiles = world.getTilesXY();
        for (auto vec : mapTiles)
        {
            for (auto vec1 : vec)
            {
                mapFile << vec1 << " ";
            }
            mapFile << endl;
        }
        mapFile << "***" << endl;
       
        auto waypoints = world.getWaypoints();
        for (auto waypoint : waypoints)
        {
            mapFile << waypoint.at(0) << " " << waypoint.at(1) << endl;
        }
        mapFile.close();
    }
